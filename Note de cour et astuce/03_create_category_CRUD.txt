# 3) Pour trier les catégories par ordre alphabétique, 
#    ajouter cette méthode dans le fichier 'src/Repository/CategoryRepository.php' : 
---------------------------------------------------------------
    public function findAll()
    {
        $builder = $this->createQueryBuilder('c');
        $builder->orderBy('c.longname', 'ASC');
        return $builder->getQuery()->getResult();
    }
---------------------------------------------------------------
